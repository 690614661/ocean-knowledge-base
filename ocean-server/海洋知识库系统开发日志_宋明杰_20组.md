```java
@PostMapping("/upload")
public CommonResp<String> upload(@RequestParam("file") MultipartFile file) {
    // 校验文件大小
    if (file.getSize() > 10 * 1024 * 1024) {
        throw new BusinessException("文件大小不能超过10MB");
    }
    
    // 校验文件类型
    String originalFilename = file.getOriginalFilename();
    String extension = originalFilename.substring(originalFilename.lastIndexOf("."));
    if (!ALLOWED_EXTENSIONS.contains(extension)) {
        throw new BusinessException("不支持的文件类型");
    }
    
    // 生成新文件名
    String newFileName = UUID.randomUUID() + extension;
    
    // 按日期创建目录
    String dateDir = LocalDate.now().format(DateTimeFormatter.ofPattern("yyyy-MM-dd"));
    String destPath = uploadPath + "/" + dateDir + "/" + newFileName;
    
    // 保存文件
    File dest = new File(destPath);
    file.transferTo(dest);
    
    // 返回访问URL
    String fileUrl = "/file/" + dateDir + "/" + newFileName;
    return new CommonResp<>(fileUrl);
}
```

#### 文件下载实现
```java
@GetMapping("/download/{dateDir}/{fileName}")
public void download(@PathVariable String dateDir, 
                     @PathVariable String fileName,
                     HttpServletResponse response) {
    File file = new File(uploadPath + "/" + dateDir + "/" + fileName);
    response.setContentType("application/octet-stream");
    response.setHeader("Content-Disposition", "attachment;filename=" + fileName);
    Files.copy(file.toPath(), response.getOutputStream());
}
```

```java
@Transactional
public void generateDailySnapshot() {
    List<Ebook> ebooks = ebookMapper.selectList(null);
    LocalDate today = LocalDate.now();
    LocalDate yesterday = today.minusDays(1);
    
    for (Ebook ebook : ebooks) {
        // 查询昨日快照
        EbookSnapshot yesterdaySnapshot = getSnapshot(ebook.getId(), yesterday);
        
        // 创建今日快照
        EbookSnapshot todaySnapshot = new EbookSnapshot();
        todaySnapshot.setEbookId(ebook.getId());
        todaySnapshot.setDate(today);
        todaySnapshot.setViewCount(ebook.getViewCount());
        todaySnapshot.setVoteCount(ebook.getVoteCount());
        
        // 计算增量
        if (yesterdaySnapshot != null) {
            todaySnapshot.setViewIncrease(
                ebook.getViewCount() - yesterdaySnapshot.getViewCount());
            todaySnapshot.setVoteIncrease(
                ebook.getVoteCount() - yesterdaySnapshot.getVoteCount());
        }
        
        save(todaySnapshot);
    }
}
```

```java
@ServerEndpoint("/ws/{userId}")
@Component
public class WebSocketServer {
    
    // 存储所有在线用户的会话
    private static ConcurrentHashMap<Long, Session> sessionMap = new ConcurrentHashMap<>();
    
    @OnOpen
    public void onOpen(Session session, @PathParam("userId") Long userId) {
        sessionMap.put(userId, session);
        broadcastOnlineCount();
    }
    
    @OnClose
    public void onClose(@PathParam("userId") Long userId) {
        sessionMap.remove(userId);
        broadcastOnlineCount();
    }
    
    @OnMessage
    public void onMessage(String message, @PathParam("userId") Long userId) {
        log.info("收到用户{}的消息: {}", userId, message);
    }
    
    // 向指定用户推送消息
    public static void sendMessage(Long userId, String message) {
        Session session = sessionMap.get(userId);
        if (session != null && session.isOpen()) {
            session.getAsyncRemote().sendText(message);
        }
    }
    
    // 广播在线用户数
    private void broadcastOnlineCount() {
        String msg = "{\"type\":\"online_count\",\"count\":" + sessionMap.size() + "}";
        sessionMap.values().forEach(session -> {
            session.getAsyncRemote().sendText(msg);
        });
    }
}
```

```java
public class XssHttpServletRequestWrapper extends HttpServletRequestWrapper {
    
    @Override
    public String getParameter(String name) {
        String value = super.getParameter(name);
        return XssFilterUtil.clean(value);
    }
    
    @Override
    public String[] getParameterValues(String name) {
        String[] values = super.getParameterValues(name);

        }
        return cleanValues;
    }
}
```

```java
@Aspect
@Component
public class RateLimitAspect {
    
    private ConcurrentHashMap<String, RateLimiter> rateLimiterMap = new ConcurrentHashMap<>();
    
    @Around("@annotation(rateLimit)")
    public Object around(ProceedingJoinPoint point, RateLimit rateLimit) throws Throwable {
        String key = RequestUtil.getIp();
        RateLimiter limiter = rateLimiterMap.computeIfAbsent(key, 
            k -> RateLimiter.create(rateLimit.permitsPerSecond()));
        
        if (!limiter.tryAcquire()) {
            throw new BusinessException("请求过于频繁，请稍后再试");
        }
        
        return point.proceed();
    }
}
```

#### 获取题目（随机抽题）
```java
public List<Question> getQuestions(Long recordId) {
    QuizRecord record = quizRecordMapper.selectById(recordId);
    if (record == null || !"in_progress".equals(record.getStatus())) {
        throw new BusinessException("答题记录无效");
    }
    
    // 随机获取题目
    List<Question> questions = questionMapper.selectList(
        new QueryWrapper<Question>()
            .eq("quiz_id", record.getQuizId())
            .orderByAsc("RAND()")
            .last("LIMIT " + record.getTotalCount())
    );
    
    // 隐藏答案和解析
    questions.forEach(q -> {
        q.setAnswer(null);
        q.setAnalysis(null);
    });
    
    return questions;
}
```

#### 提交答卷与判分
```java
@Transactional
public QuizRecord submitQuiz(Long recordId, Map<Long, String> userAnswers) {
    QuizRecord record = quizRecordMapper.selectById(recordId);
    if (record == null || !"in_progress".equals(record.getStatus())) {
        throw new BusinessException("答题记录无效");
    }
    
    // 获取所有题目和正确答案
    List<Question> questions = questionMapper.selectList(
        new QueryWrapper<Question>().eq("quiz_id", record.getQuizId())
    );
    
    int score = 0;
    int correctCount = 0;
    
    // 判分
    for (Question question : questions) {
        String userAnswer = userAnswers.get(question.getId());
        boolean isCorrect = question.getAnswer().equals(userAnswer);
        
        if (isCorrect) {
            correctCount++;
            score += question.getScore();
        }
        
        // 保存答题详情
        QuizRecordDetail detail = new QuizRecordDetail();
        detail.setRecordId(recordId);
        detail.setQuestionId(question.getId());
        detail.setUserAnswer(userAnswer);
        detail.setIsCorrect(isCorrect ? 1 : 0);
        quizRecordDetailMapper.insert(detail);
    }
    
    // 更新答题记录
    record.setScore(score);
    record.setCorrectCount(correctCount);
    record.setEndTime(new Date());
    record.setStatus("completed");
    quizRecordMapper.updateById(record);
    
    return record;
}
```

```java
public List<QuizRecord> getRanking(Long quizId, Integer limit) {
    return quizRecordMapper.selectList(
        new QueryWrapper<QuizRecord>()
            .eq("quiz_id", quizId)
            .eq("status", "completed")
            .orderByDesc("score")
            .orderByAsc("end_time") // 分数相同，按完成时间排序
            .last("LIMIT " + limit)
    );
}
```

**问题1**：随机抽题性能慢

**解决**：后续可增加定时保存答案功能，支持断点续答
